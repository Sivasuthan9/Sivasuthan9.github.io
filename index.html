package com.speechify; import com.fasterxml.jackson.databind.ObjectMapper; import com.fasterxml.jackson.databind.JsonNode; import com.fasterxml.jackson.databind.node.ObjectNode; import com.fasterxml.jackson.databind.node.ArrayNode; import java.io.File; import java.io.IOException; import java.util.ArrayList; import java.util.List; import java.util.concurrent.CompletableFuture; public class UserRepository { private static final String DB_FILE = "db.json"; private static final String USERS_PROPERTY = "users"; private final ObjectMapper objectMapper; private final LRUCache cache; public UserRepository(ObjectMapper objectMapper, LRUCache cache) { this.objectMapper = objectMapper; this.cache = cache; } public UserRepository() { this.objectMapper = new ObjectMapper(); this.cache = new LRUCache<>(100); // Default cache size } public CompletableFuture saveUser(User user) { return CompletableFuture.supplyAsync(() -> { try { File dbFile = new File(DB_FILE); ObjectNode root = dbFile.exists() ? (ObjectNode) objectMapper.readTree(dbFile) : objectMapper.createObjectNode(); ArrayNode users = root.has(USERS_PROPERTY) ? (ArrayNode) root.get(USERS_PROPERTY) : root.putArray(USERS_PROPERTY); // Check if user exists and update, else add new boolean userExists = false; for (int i = 0; i < users.size(); i++) { JsonNode userNode = users.get(i); if (userNode.has("id") && userNode.get("id").asText().equals(user.getId())) { users.set(i, objectMapper.valueToTree(user)); userExists = true; break; } } if (!userExists) { users.add(objectMapper.valueToTree(user)); } objectMapper.writeValue(dbFile, root); cache.put(user.getId(), user); // Update cache cache.put(user.getEmail(), user); // Also cache by email return true; } catch (IOException e) { return false; } }); } public CompletableFuture> getAllUsers() { return CompletableFuture.supplyAsync(() -> { try { File dbFile = new File(DB_FILE); if (!dbFile.exists()) { return new ArrayList<>(); } JsonNode root = objectMapper.readTree(dbFile); JsonNode users = root.get(USERS_PROPERTY); if (users == null) { return new ArrayList<>(); } List userList = new ArrayList<>(); for (JsonNode userNode : users) { User user = objectMapper.treeToValue(userNode, User.class); userList.add(user); // Cache individual users cache.put(user.getId(), user); cache.put(user.getEmail(), user); } return userList; } catch (IOException e) { return new ArrayList<>(); } }); } public CompletableFuture getUserByEmail(String email) { // Check cache first User cachedUser = cache.get(email); if (cachedUser != null) { return CompletableFuture.completedFuture(cachedUser); } return CompletableFuture.supplyAsync(() -> { try { File dbFile = new File(DB_FILE); if (!dbFile.exists()) { return null; } JsonNode root = objectMapper.readTree(dbFile); JsonNode users = root.get(USERS_PROPERTY); if (users == null) { return null; } for (JsonNode userNode : users) { if (userNode.has("email") && userNode.get("email").asText().equals(email)) { User user = objectMapper.treeToValue(userNode, User.class); cache.put(user.getId(), user); cache.put(email, user); return user; } } return null; } catch (IOException e) { return null; } }); } public CompletableFuture getUserById(String id) { User cachedUser = cache.get(id); if (cachedUser != null) { return CompletableFuture.completedFuture(cachedUser); } // Fallback to searching through all users return getAllUsers().thenApply(users -> users.stream() .filter(user -> user.getId().equals(id)) .findFirst() .orElse(null) ); } }
